#!/bin/bash

#export LD_LIBRARY_PATH=/home/jan/Qt/6.10.2/gcc_64/lib:$LD_LIBRARY_PATH
#export LD_LIBRARY_PATH=/home/jan/qwt-6.3.0/lib
#export PATH=/home/jan/Qt/6.10.2/gcc_64/bin:$PATH

ls -lh .
rm -rf *_autogen
mkdir -p ./appdir/usr/bin
mkdir -p ./appdir/usr/share/applications
cp linux-bin/ft8-module  appdir/usr/bin/
 
cp ft8-module.desktop appdir/usr/share/applications
cp ft8-module.png appdir/ft8-module.png
touch appdir/ft8-module.png
mkdir -p ./appdir/usr/share/icons/hicolor/256x256/apps/
cp ft8-module.png appdir/usr/share/icons/hicolor/256x256/apps/
  
wget -c "https://github.com/probonopd/linuxdeployqt/releases/download/continuous/linuxdeployqt-continuous-x86_64.AppImage" 
chmod a+x linuxdeployqt*.AppImage
unset QTDIR; unset QT_PLUGIN_PATH ; unset LD_LIBRARY_PATH
./linuxdeployqt*.AppImage ./appdir/usr/share/applications/* -bundle-non-qt-libs -no-translations
./linuxdeployqt*.AppImage --appimage-extract
find appdir/usr/plugins/ -type f -exec squashfs-root/usr/bin/patchelf --set-rpath '$ORIGIN/../../lib' {} \;
chmod a+x appimage/* ; rm appdir/AppRun ; cp appimage/* appdir/
export PATH=squashfs-root/usr/bin/:$PATH
squashfs-root/usr/bin/appimagetool $(readlink -f ./appdir/)
find ./appdir -executable -type f -exec ldd {} \; | grep " => /usr" | cut -d " " -f 2-3 | sort | uniq
:
